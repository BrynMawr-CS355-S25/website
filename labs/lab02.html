Lab exercises for January 30th

The goals for this assignment are:

  • Working with C strings

  • Working with processes

#include <stdio.h> #include <string.h> #define delims " \t"

void print(char list) { printf("["); for (char ptr = list; *ptr != NULL; ptr++) { printf("%s, ", *ptr); } printf("NULL]\n"); }

int main() { char line[1024]; strcpy(line, "ls -l -a");

    int n = 0;
    char* list[128]; // max of 128 words can be put in the list
    char* token = strtok(line, delims);
    while (token)
    {
        list[n++] = token;
        token = strtok(NULL, delims);
    }
    list[n] = NULL;
    print(list);
}
